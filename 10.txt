#include <iostream>
#include <fstream>
#include <string>

using namespace std;

class Point{
public:
    int x, y, z;
    Point(){
        x = y = z = 0;
    }
    Point (int x, int y, int z){
        this->x = x;
        this->y = y;
        this->z = z;
    }
    void Print (){
        cout << "X:"<< x<< "Y:"<< y<< "Z:"<< z << endl;
    }
};
int main (){
    string path = "test.txt";
    Point point (1, 2, 5);
    ofstream fout;
    fout.open(path, ofstream::app);

    fout.write((char *) &point, sizeof(Point));
    fout.close();
    return 0;
}
