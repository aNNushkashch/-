#include <iostream>
using namespace std;

int bubble_sort(int arr[], int size){
    int buff = 0;
    for (int i = 0; i < size - 1; i++){
        for (int j = size - 1; j > i; j--){
            if (arr[j] < arr[j - 1]){
                buff = arr[j - 1];
                arr[j - 1] = arr[j];
                arr[j] = buff;
            }
        }
    }
    cout<<"Rezult array: ";
    for (int i=0; i<size; i++){
            cout<<arr[i]<<" ";
    }
}

int main() {
    int *arr;
    int size;
    cout<<"Enter size: \n";
    cin >> size;

    cout<<"Enter array: \n";
    arr=new int[size];
    for(int i=0;i < size; i++)
        cin >> arr[i];

    bubble_sort(arr, size);

}
